% --- SECTION: MULTIPLE CHOICE ---
\noindent\textbf{\large\color{workbookteal} Choose the correct answer. Write its number in the parentheses.}

\begin{enumerate}[label=\arabic*., leftmargin=1cm, itemsep=15pt]
    \item Which of the following shows the correct numeral for seventy-two thousand, eight hundred, forty-five?
    \begin{multicols}{2}
    \begin{enumerate}[label=(\arabic*)]
        \item 72,845
        \item 72,854
        \item 78,245
        \item 78,254 \hfill ( \showanswer{1} )
    \end{enumerate}
    \end{multicols}
    \tip{Read carefully: "Seventy-two thousand" = 72,000. "Eight hundred, forty-five" = 845.}

    \item The digit 8 in 28,095 stands for \blankline{1.5cm}.
    \begin{multicols}{2}
    \begin{enumerate}[label=(\arabic*)]
        \item 8 ten thousands
        \item 8 thousands
        \item 8 hundreds
        \item 8 tens \hfill ( \showanswer{2} )
    \end{enumerate}
    \end{multicols}
    \tip{2\underline{8},095 is in the thousands place. Value = 8,000.}

    \item Which of the following is \textbf{not} a common multiple of 8 and 6?
    \begin{multicols}{2}
    \begin{enumerate}[label=(\arabic*)]
        \item 18
        \item 24
        \item 72
        \item 144 \hfill ( \showanswer{1} )
    \end{enumerate}
    \end{multicols}
    \solution{Multiples of 6: 6, 12, \textbf{18}, 24... \\
    Multiples of 8: 8, 16, 24... \\
    18 is a multiple of 6 but not 8.}

    \item Round each number to the nearest ten and estimate the value of $1,987 + 5,248$.
    \begin{multicols}{2}
    \begin{enumerate}[label=(\arabic*)]
        \item 7,220
        \item 7,230
        \item 7,240
        \item 7,250 \hfill ( \showanswer{3} )
    \end{enumerate}
    \end{multicols}
    \solution{$1,987 \approx 1,990$. $5,248 \approx 5,250$. \\
    $1,990 + 5,250 = 7,240$.}

    \item Which of the following is a common factor of 28 and 36?
    \begin{multicols}{2}
    \begin{enumerate}[label=(\arabic*)]
        \item 3
        \item 4
        \item 6
        \item 8 \hfill ( \showanswer{2} )
    \end{enumerate}
    \end{multicols}
    \solution{Factors of 28: 1, 2, \textbf{4}, 7, 14, 28. \\
    Factors of 36: 1, 2, 3, \textbf{4}, 6, 9... \\
    4 is in both lists.}

    \item 9,050, \blankline{1.5cm}, 7,030, 6,020, 5,010. What is the missing number in the pattern?
    \begin{multicols}{2}
    \begin{enumerate}[label=(\arabic*)]
        \item 8,020
        \item 8,030
        \item 8,040
        \item 8,050 \hfill ( \showanswer{3} )
    \end{enumerate}
    \end{multicols}
    \solution{Pattern is decreasing by 1,010. \\
    $7,030 + 1,010 = 8,040$.}

    \item Which of the following has the greatest value?
    \begin{multicols}{2}
    \begin{enumerate}[label=(\arabic*)]
        \item 2,000 less than 10,000
        \item 2,000 less than 8,000
        \item 2,000 more than 1,000
        \item 2,000 more than 800 \hfill ( \showanswer{1} )
    \end{enumerate}
    \end{multicols}
    \solution{(1) $10,000 - 2,000 = 8,000$. \\
    (2) $8,000 - 2,000 = 6,000$. \\
    (3) $1,000 + 2,000 = 3,000$. \\
    (4) $800 + 2,000 = 2,800$.}

    \item Which of the following shows the first four multiples of 7?
    \begin{multicols}{2}
    \begin{enumerate}[label=(\arabic*)]
        \item 7, 14, 20, 27
        \item 7, 14, 21, 28
        \item 7, 14, 28, 35
        \item 7, 14, 28, 42 \hfill ( \showanswer{2} )
    \end{enumerate}
    \end{multicols}
    \solution{$7 \times 1 = 7$, $7 \times 2 = 14$, $7 \times 3 = 21$, $7 \times 4 = 28$.}

    \item $49,753 = \blankline{1.5cm} + 9$ thousands $+ 7$ hundreds $+ 5$ tens $+ 3$ ones
    \begin{multicols}{2}
    \begin{enumerate}[label=(\arabic*)]
        \item 400 ten thousands
        \item 40 ten thousands
        \item 4 ten thousands
        \item 4 thousands \hfill ( \showanswer{3} )
    \end{enumerate}
    \end{multicols}
    \solution{The digit 4 is in the ten thousands place. \\
    Value = 40,000 = 4 ten thousands. \\
    (Note: 40 ten thousands would be 400,000).}

    \item Estimate the value of $404 \times 9$.
    \begin{multicols}{2}
    \begin{enumerate}[label=(\arabic*)]
        \item 3,600
        \item 3,636
        \item 3,645
        \item 4,000 \hfill ( \showanswer{1} )
    \end{enumerate}
    \end{multicols}
    \solution{$404 \approx 400$. \\
    $400 \times 9 = 3,600$.}
\end{enumerate}

    \item In 94,857, the digit 4 is in the \showanswer{thousands} place. \hfill \blankline{4cm}

    \item Arrange these numbers in ascending order.
    \\[10pt] 15,050, \quad 15,005, \quad 15,500
    \\[10pt] \showanswer{15,005, \quad 15,050, \quad 15,500}
    \tip{Compare digits from left to right. All start with 15. Next digits: 0 vs 0 vs 5. Then look at tens/ones.}

    \item Round each number to the nearest ten and estimate its value.
    \\[10pt] $559 + 19 + 942 = $ \showanswer{1,520} \hfill \blankline{4cm}
    \solution{$559 \approx 560$. $19 \approx 20$. $942 \approx 940$. \\
    $560 + 20 + 940 = 580 + 940 = 1,520$.}

    \item The seventh multiple of 9 is \showanswer{63}. \hfill \blankline{4cm}
    \solution{$7 \times 9 = 63$.}

    \item Round 89,091 to the nearest hundred. \hfill \showanswer{89,100} \blankline{2cm}
    \solution{Digit in tens place is 9 ($>5$), so round up the hundreds digit (0 becomes 1).}

    \item 50 thousands + 90 tens + 7 ones = \showanswer{50,907} \hfill \blankline{4cm}
    \solution{50 thousands = 50,000. \\
    90 tens = 900 (Not 90 in the tens place!). \\
    7 ones = 7. \\
    $50,000 + 900 + 7 = 50,907$.}

    \item List all the factors of 45. \hfill \showanswer{1, 3, 5, 9, 15, 45} \blankline{0cm}
    \solution{$1 \times 45$, $3 \times 15$, $5 \times 9$.}

    \item Add 18,360 and 2,598. The digit \showanswer{0} is in the thousands place. \hfill \blankline{4cm}
    \solution{$18,360 + 2,598 = 20,958$. \\
    The digit in the natural thousands place (4th from right) is 0.}

    \item List the first two common multiples of 4 and 6. \hfill \showanswer{12, 24} \blankline{2cm}
    \solution{Multiples of 4: 4, 8, \textbf{12}, 16, 20, \textbf{24}... \\
    Multiples of 6: 6, \textbf{12}, 18, \textbf{24}...}
\end{enumerate}
